/**
 * @fileoverview Abstract class of menu elements.
 *
 * @author volker.sorge@gmail.com (Volker Sorge)
 */
/// <reference path="abstract_navigatable.ts" />
/// <reference path="html_classes.ts" />
var ContextMenu;
(function (ContextMenu) {
    class MenuElement extends ContextMenu.AbstractNavigatable {
        /**
         * @constructor
         * @extends {AbstractNavigatable}
         * @implements {Element}
         */
        constructor(...args) {
            super(...args);
        }
        /**
         * Adds a attributes and values to the HTML element.
         * @param {Object.<string, string>} attributes A dictionary of attributes.
         */
        addAttributes(attributes) {
            for (let attr in attributes) {
                this.html.setAttribute(attr, attributes[attr]);
            }
        }
        /**
         * @override
         */
        getHtml() {
            if (!this.html) {
                this.generateHtml();
            }
            return this.html;
        }
        /**
         * @override
         */
        setHtml(html) {
            this.html = html;
            this.addEvents(html);
        }
        /**
         * @override
         */
        generateHtml() {
            //// TODO: Make this DOM independent!
            let html = document.createElement('div');
            html.classList.add(this.className);
            html.setAttribute('role', this.role);
            this.setHtml(html);
        }
        /**
         * @override
         */
        focus() {
            let html = this.getHtml();
            html.setAttribute('tabindex', '0');
            html.focus();
        }
        /**
         * @override
         */
        unfocus() {
            let html = this.getHtml();
            if (html.hasAttribute('tabindex')) {
                html.setAttribute('tabindex', '-1');
            }
            html.blur();
        }
    }
    ContextMenu.MenuElement = MenuElement;
})(ContextMenu || (ContextMenu = {}));
